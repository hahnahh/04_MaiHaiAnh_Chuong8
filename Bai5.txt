n = int(input("Nhập n: "))

# S1 = 1 + 2 + ... + n
S1 = sum(range(1, n+1))
# S2 = 1*2*3*...*(n-1)
S2 = 1
for i in range(1, n):
    S2 *= i
# S3 = 1 - 1/2 + 1/3 - 1/4 + ... + ((-1)**n)/n
S3 = sum(((-1)**k)/k for k in range(1, n+1))

# S4 = tổng k/(k+2) từ k=0 đến n
S4 = sum(k/(k+2) for k in range(0, n+1))

print("S1 =", S1)
print("S2 =", S2)
print("S3 =", S3)
print("S4 =", S4)
